💻 Usage Example

Get started with a simple example:

// Import the Leaflet wrapper import LeafletWrapper from './leaflet-wrapper.js'; // Create a map instance const mapWrapper = new LeafletWrapper('map-container', { center: [51.505, -0.09], zoom: 13 }); // Add a marker with popup mapWrapper.addMarker([51.5, -0.09], 'Hello World!'); // Squirrel integration const mapAtome = atome({ tag: 'div', id: 'my-map' }); const leafletMap = mapAtome.leaflet({ center: [40.7128, -74.0060], zoom: 10 });
